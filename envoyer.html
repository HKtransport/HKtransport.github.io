function calculer() { let depart = document.getElementById("depart").value; let arrivee = document.getElementById("arrivee").value; let poids = Number(document.getElementById("poids").value); let option = document.getElementById("option").value; if (poids <= 0) { alert("Veuillez entrer un poids valide."); return; } // Prix du poids let prixPoids = 500; // moins de 1kg = 500 CFA if (poids > 1) { prixPoids = 500 + (Math.ceil(poids - 1) * 500); } // Prix fixe pour l'axe Saint-Louis → Dakar const ordre = ["Saint-Louis", "Louga", "Thiès", "Dakar"]; let distance = Math.abs(ordre.indexOf(depart) - ordre.indexOf(arrivee)); let prixDistance = distance * 1500; // Option express let prixExpress = (option === "express") ? 1500 : 0; // Total let prix = prixPoids + prixDistance + prixExpress; // Affichage document.getElementById("resultat").innerHTML = "Prix total : " + prix + " CFA"; // Bouton paiement const btn = document.getElementById("btnPaiement"); btn.style.display = "block"; btn.onclick = function() { window.location.href = "paiement.html?prix=" + prix; }; }
